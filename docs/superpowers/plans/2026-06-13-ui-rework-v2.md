# ccw UI Rework v2 (Hybrid Atelier + Glass) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace v1's cyan-blue modern dashboard with a hybrid visual system — warm Atelier (krem, rounded, friendly) as the base + Glass (frosted + Champagne gradient) reserved for hero and key callouts only.

**Architecture:** Keep v1's component library primitives (Radix, cva, cmdk, lucide) and stack (React 19, Vite 7, TypeScript, Tailwind 4 with `@theme inline` + OKLCH tokens). Replace token set, add Fraunces display font, drop persistent sidebar in favor of inline-text top-bar nav, add a glass surface utility class, then re-style every page to use the new system. Forms and tables stay dense and readable; glass appears only on hero, success toast, and one callout per page at most.

**Tech Stack:** React 19, Vite 7, Tailwind 4 (`@theme inline`, OKLCH), Radix UI primitives, cva, cmdk, lucide-react, @monaco-editor/react, i18next, sonner, Fraunces (Google Font → local woff2 in `public/fonts/`), Inter (already), JetBrains Mono (already).

**Reference spec:** `docs/superpowers/specs/2026-06-13-ui-rework-v2-design.md` (sections 1–11).

// 9 warm swatches. No neon, no purple. Each is a 2-stop linear gradient.
export const SWATCHES: { name: string; from: string; to: string }[] = [
  { name: "honey",  from: "oklch(0.92 0.05 80)",  to: "oklch(0.78 0.09 70)" },
  { name: "amber",  from: "oklch(0.85 0.09 65)",  to: "oklch(0.65 0.13 50)" },

  { name: "moss",   from: "oklch(0.78 0.06 130)", to: "oklch(0.55 0.1 140)" },
  { name: "sage",   from: "oklch(0.85 0.04 150)", to: "oklch(0.65 0.07 155)" },
  { name: "ocean",  from: "oklch(0.75 0.06 220)", to: "oklch(0.5 0.1 230)" },
  { name: "dusk",   from: "oklch(0.65 0.06 280)", to: "oklch(0.4 0.08 290)" },
  { name: "graphite", from: "oklch(0.55 0.005 60)", to: "oklch(0.35 0.008 60)" },
];

function hash(input: string): number {
  let h = 0;
  for (let i = 0; i < input.length; i++) h = (h * 31 + input.charCodeAt(i)) | 0;
  return Math.abs(h);
}

export function swatchFor(seed: string): { from: string; to: string; name: string } {
  return SWATCHES[hash(seed) % SWATCHES.length];
}
